00042 /* Starts up the external 48 MHz oscillator and configures
00043  * the processor to run at 24 MHz.
00044  */
00045 void boardClockInit()
00046 {
00047     // OSC_PD=0: Power up both high-speed oscillators: XOSC and HS RCOSC.
00048     SLEEP &= ~0x04;
00049 
00050     // Wait until the high speed crystal oscillator is stable (SLEEP.XOSC_STB=1)
00051     // This condition is required to use the radio or USB.
00052     // According to Table 6.4.2, the waiting should take about 650 microseconds.
00053     while(!(SLEEP & 0x40));
00054 
00055     // OSC32K = 1: Use low power 32kHz oscillator for the 32 kHz signal.
00056     // OSC=0: Select high speed crystal oscillator for system clock (24 MHz).
00057     // TICKSPD=000: Set the timer tick speed to its fastest possible value.
00058     // CLKSPD=000: Set system clock speed to its fastest possible value (24 MHz).
00059     //    This is required for using the Forward Error Correction radio feature (which we don't use anymore).
00060     CLKCON = 0x80;
00061 
00062     // Power down the HS RCOSC (the one that is not currently selected by
00063     // CLKCON.OSC).
00064     SLEEP |= 0x04;
00065 
00066     // Enable pre-fetching of instructions from flash,
00067     // which makes the code execute much faster.
00068     MEMCTR = 0;
00069 }

00071 void boardIoInit()
00072 {
00073     P2DIR = 0;           // Make all the Port 2 pins be inputs.
00074     P2 = 0b00000110;     // P2_1 = 1: drive the red LED line high LATER (when LED_RED(1) is called)
00075                          // P2_2 = 1: drive the yellow LED line high LATER (when LED_YELLOW(1) is called)
00076                          // P2_4 = 0: drive the VBUS_IN/GREEN_LED line low LATER (when LED_GREEN(1) is called)
00077     P2INP = 0b10011001;  // Pull down LED pins (P2_2, P2_1), and tristate the other Port 2 pins.
00078 }

00080 /* Checks to see if USB is connected (VBUS_IN line is high).
00081  * The check is only performed if it has not been performed within the last 25 milliseconds.
00082  * (USB spec says we have to detect power loss within 100 ms).
00083  * This function updates the bit variable vbusHigh. */
00084 static void boardDetectVbus()
00085 {
00086     static uint8 lastCheck = 128;
00087     if ((uint8)(getMs() - lastCheck) > 25)
00088     {
00089         BIT savedState = (P2DIR >> 4) & 1;
00090         if (savedState == 0)
00091         {
00092             P2DIR |= (1<<4);       // Drive the VBUS_IN low
00093             delayMicroseconds(2);
00094         }
00095         P2INP &= ~(1<<4);          // Set input mode to pull-down
00096         P2DIR &= ~(1<<4);          // Make the line an input.
00097         delayMicroseconds(1);
00098 
00099         vbusHighBit = P2_4;           // Measure the voltage.
00100 
00101         P2INP |= (1<<4);           // Set input mode to tristate.
00102         if (savedState)
00103         {
00104             P2DIR |= (1<<4);  // LED was on previously so turn it back on.
00105         }
00106 
00107         lastCheck = getMs();
00108     }
00109 }

00111 void boardStartBootloader()
00112 {
00113     EA = 0;             // Disable interrupts.
00114 
00115     DMAARM = 0x9F;      // Disarm all DMA channels.
00116 
00117     delayMs(10);        // Probably not necessary anymore.

00120     P1DIR = 0;          // the bootloader anyway.  We might as well do it now so that any devices
00121     P2DIR = 0;          // such as motors stop running right away.  This also signals to the USB host
00122                         // that we are disconnecting.
00123 
00124     delayMs(100);       // Delay to give the USB host a chance to detect that we disconnected.
00125     __asm ljmp 6 __endasm;
00126 }
