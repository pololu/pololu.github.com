00391 // Implements all the control transfers that are required by Appendix G of HID 1.11.
00392 void usbCallbackSetupHandler(void)
00393 {
00394     static XDATA uint8 response;
00395 
00396     if ((usbSetupPacket.bmRequestType & 0x7F) != 0x21)   // Require Type==Class and Recipient==Interface.
00397         return;
00398 
00399     switch(usbSetupPacket.bRequest)
00400     {
00401     // required
00402     case HID_REQUEST_GET_REPORT:
00403         if ((usbSetupPacket.wValue >> 8) != HID_REPORT_TYPE_INPUT)
00404             return;
00405 
00406         switch (usbSetupPacket.wIndex)
00407         {
00408         case HID_KEYBOARD_INTERFACE_NUMBER:
00409             usbControlRead(sizeof(usbHidKeyboardInput), (uint8 XDATA *)&usbHidKeyboardInput);
00410             return;
00411 
00412         case HID_MOUSE_INTERFACE_NUMBER:
00413             usbControlRead(sizeof(usbHidMouseInput), (uint8 XDATA *)&usbHidMouseInput);
00414             return;
00415 
00416         case HID_JOYSTICK_INTERFACE_NUMBER:
00417             usbControlRead(sizeof(usbHidJoystickInput), (uint8 XDATA *)&usbHidJoystickInput);
00418             return;
00419 
00420         default:
00421             // unrecognized interface - stall
00422             return;
00423         }
00424 
00425     // required for devices with Output reports
00426     case HID_REQUEST_SET_REPORT:
00427         if (usbSetupPacket.wIndex == HID_KEYBOARD_INTERFACE_NUMBER)
00428         {
00429             usbControlWrite(sizeof(usbHidKeyboardOutput), (uint8 XDATA *)&usbHidKeyboardOutput);
00430         }
00431         return;
00432 
00433     // required for keyboards
00434     case HID_REQUEST_GET_IDLE:
00435         if (usbSetupPacket.wIndex == HID_KEYBOARD_INTERFACE_NUMBER)
00436         {
00437             response = hidKeyboardIdleDuration / 4; // value in request is in units of 4 ms
00438             usbControlRead(1, (uint8 XDATA *)&response);
00439         }
00440         return;
00441 
00442     // required for keyboards
00443     case HID_REQUEST_SET_IDLE:
00444         if (usbSetupPacket.wIndex == HID_KEYBOARD_INTERFACE_NUMBER)
00445         {
00446             hidKeyboardIdleDuration = (usbSetupPacket.wValue >> 8) * 4; // value in request is in units of 4 ms
00447             usbControlAcknowledge();
00448         }
00449         return;
00450 
00451 
00452     // required for boot devices
00453     case HID_REQUEST_GET_PROTOCOL:
00454         switch (usbSetupPacket.wIndex)
00455         {
00456         case HID_KEYBOARD_INTERFACE_NUMBER:
00457             response = hidKeyboardProtocol;
00458             usbControlRead(1, (uint8 XDATA *)&response);
00459             return;
00460 
00461         case HID_MOUSE_INTERFACE_NUMBER:
00462             response = hidMouseProtocol;
00463             usbControlRead(1, (uint8 XDATA *)&response);
00464             return;
00465 
00466         default:
00467             // unrecognized interface - stall
00468             return;
00469         }
00470 
00471     // required for boot devices
00472     case HID_REQUEST_SET_PROTOCOL:
00473         switch (usbSetupPacket.wIndex)
00474         {
00475         case HID_KEYBOARD_INTERFACE_NUMBER:
00476             hidKeyboardProtocol = usbSetupPacket.wValue;
00477             break;
00478 
00479         case HID_MOUSE_INTERFACE_NUMBER:
00480             hidMouseProtocol = usbSetupPacket.wValue;
00481             break;
00482 
00483         default:
00484             // unrecognized interface - stall
00485             return;
00486         }
00487         usbControlAcknowledge();
00488         return;
00489 
00490     default:
00491         // unrecognized request - stall
00492         return;
00493     }
00494 }

00551 void usbHidService(void)
00552 {
00553     static uint16 XDATA hidKeyboardLastReportTime = 0;
00554 
00555     usbPoll();
00556 
00557     if (usbDeviceState != USB_STATE_CONFIGURED)
00558     {
00559         // We have not reached the Configured state yet, so we should not be touching the non-zero endpoints.
00560         return;
00561     }
00562 
00563     USBINDEX = HID_KEYBOARD_ENDPOINT;
00564     // Check if keyboard input has been updated OR if the idle period is nonzero and has expired.
00565     if ((usbHidKeyboardInputUpdated || (hidKeyboardIdleDuration && ((uint16)(getMs() - hidKeyboardLastReportTime) > hidKeyboardIdleDuration))) && !(USBCSIL & USBCSIL_INPKT_RDY))
00566     {
00567         usbWriteFifo(HID_KEYBOARD_ENDPOINT, sizeof(usbHidKeyboardInput), (uint8 XDATA *)&usbHidKeyboardInput);
00568         USBCSIL |= USBCSIL_INPKT_RDY;
00569         usbHidKeyboardInputUpdated = 0; // reset updated flag
00570         hidKeyboardLastReportTime = getMs();
00571     }
00572 
00573     USBINDEX = HID_MOUSE_ENDPOINT;
00574     // Check if mouse input has been updated.
00575     if (usbHidMouseInputUpdated && !(USBCSIL & USBCSIL_INPKT_RDY)) {
00576         usbWriteFifo(HID_MOUSE_ENDPOINT, sizeof(usbHidMouseInput), (uint8 XDATA *)&usbHidMouseInput);
00577         USBCSIL |= USBCSIL_INPKT_RDY;
00578         usbHidMouseInputUpdated = 0; // reset updated flag
00579     }
00580 
00581     USBINDEX = HID_JOYSTICK_ENDPOINT;
00582     // Check if joystick input has been updated.
00583     if (usbHidJoystickInputUpdated && !(USBCSIL & USBCSIL_INPKT_RDY)) {
00584         usbWriteFifo(HID_JOYSTICK_ENDPOINT, sizeof(usbHidJoystickInput), (uint8 XDATA *)&usbHidJoystickInput);
