00021 // For highest throughput, we want to send as much data in each packet
00022 // as possible.  But for lower latency, we sometimes need to send packets
00023 // that are NOT full.
00024 // This library will only send non-full packets if the number of packets
00025 // currently queued to be sent is small.  Specifically, that number must
00026 // not exceed TX_QUEUE_THRESHOLD.
00027 // A higher threshold means that there will be more under-populated packets
00028 // at the beginning of a data transfer (which is bad), but slightly reduces
00029 // the importance of calling radioComTxService often (which can be good).
00030 #define TX_QUEUE_THRESHOLD  1

00041 static void receiveMorePackets(void)
00042 {
00043     uint8 XDATA * packet;
00044 
00045     if (rxBytesLeft != 0)
00046     {
00047         // There are bytes available.  The higher-level code should
00048         // call radioComRxReceiveByte to get them.
00049         return;
00050     }
00051 
00052     if (WAITING_TO_REPORT_RX_SIGNALS)
00053     {
00054         // The higher-level code needs to call radioComRxSignals before
00055         // we feed it any more data.
00056         return;
00057     }
00058 
00059     // Each iteration of this loop processes one packet received on the radio.
00060     // This loop stops when we are out of packets or when we received a packet
00061     // that contains some information that the higher-level code needs to process.
00062     while(packet = radioLinkRxCurrentPacket())
00063     {
00064         switch(radioLinkRxCurrentPayloadType())
00065         {
00066         case PAYLOAD_TYPE_DATA:
00067             // We received some data.  Populate rxPointer and rxBytesLeft.
00068             // The data can be retreived with radioComRxAvailable and racioComRxReceiveByte().
00069 
00070             // Assumption: radioLink doesn't ever return zero-length packets,
00071             // so rxBytesLeft is non-zero now and we don't have to worry about
00072             // discard zero-length packets in radio_com.c.
00073             rxBytesLeft = packet[0];  // Read the packet length.
00074             rxPointer = packet+1;              // Make rxPointer point to the data.
00075             return;
00076 
00077         case PAYLOAD_TYPE_CONTROL_SIGNALS:
00078             // We received a command to set the control signals.
00079             radioComRxSignals = packet[1];
00080 
00081             radioLinkRxDoneWithPacket();
00082 
00083             if (WAITING_TO_REPORT_RX_SIGNALS)
00084             {
00085                 // The higher-level code has not seen these values for the control
00086                 // signals yet, so stop processing packets.
00087                 // The higher-level code can access these values by calling radioComRxControlSignals().
00088                 return;
00089             }
00090 
00091             // It was a redundant command so don't do anything special.
00092             // Keep processing packets.
00093             break;
00094         }
00095     }
00096 }
