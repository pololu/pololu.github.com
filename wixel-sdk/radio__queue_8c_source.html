00001 /* radio_queue.c:
00002  *  This layer builds on top of radio_mac.c to provide a mechanism for queueing
00003  *  RF packets to be sent and packets that are received by this device. It does
00004  *  not ensure reliability or specify the format of the packets, except that the
00005  *  first byte of the packet must contain its length.
00006  *
00007  *  This layer does not transmit packets as quickly as possible; instead, it
00008  *  listens for incoming packets for a random interval of 1-4 ms between sending
00009  *  packets.
00010  *
00011  *  This layer defines the RF packet memory buffers used, and controls access to
00012  *  those buffers.
00013  *
00014  *  Radio_queue is essentially a stripped-down version of the radio_link
00015  *  library, so radio_link is a good alternative if you want a more specialized
00016  *  implementation with more features.
00017  */

00030 #define RADIO_MAX_PACKET_SIZE  (RADIO_QUEUE_PAYLOAD_SIZE)

00032 #define RADIO_QUEUE_PACKET_LENGTH_OFFSET 0

00034 /*  rxPackets:
00035  *  We need to be prepared at all times to receive a full packet from another
00036  *  party, even if we cannot give it to the main loop.  Therefore, we need (at
00037  *  least) THREE buffers, so that two can be owned by the main loop while
00038  *  another is owned by the ISR and ready to receive the next packet.
00039  *
00040  *  If a packet is received and the main loop still owns the other two buffers,
00041  *  we discard it.
00042  *
00043  *  Ownership of the RX packet buffers is determined from radioQueueRxMainLoopIndex and radioQueueRxInterruptIndex.
00044  *  The main loop owns all the buffers from radioQueueRxMainLoopIndex to radioQueueRxInterruptIndex-1 inclusive.
00045  *  If the two indices are equal, then the main loop owns nothing.  Here are three examples:
00046  *
00047  *  radioQueueRxMainLoopIndex | radioQueueRxInterruptIndex | Buffers owned by main loop.
00048  *                0 |                0 | None
00049  *                0 |                1 | rxBuffer[0]
00050  *                0 |                2 | rxBuffer[0 and 1]
00051  */
00052 #define RX_PACKET_COUNT  3
00053 static volatile uint8 XDATA radioQueueRxPacket[RX_PACKET_COUNT][1 + RADIO_MAX_PACKET_SIZE + 2];  // The first byte is the length.
00054 static volatile uint8 DATA radioQueueRxMainLoopIndex = 0;   // The index of the next rxBuffer to read from the main loop.
00055 static volatile uint8 DATA radioQueueRxInterruptIndex = 0;  // The index of the next rxBuffer to write to when a packet comes from the radio.
00056

00061 static volatile uint8 DATA radioQueueTxInterruptIndex = 0;  // The index of the current txPacket we are trying to send on the radio.
