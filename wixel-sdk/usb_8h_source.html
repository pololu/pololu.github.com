00226 /* usbSuspendMode:  Direct access to this bit is provided for applications that
00227  * need to use the P0 interrupt and want USB suspend mode to work.  If you don't
00228  * fall in to that category, please don't use this bit directly: instead you
00229  * should call usbSuspended() because that function is less likely to change in
00230  * future versions.  If you want to have a P0 interrupt AND USB Suspend, you
00231  * should write your ISR like this:
00232  *
00233  * ISR(P0INT, 1)
00234  * {
00235  *     // Handle the P0 flags you care about here, but you might want to
00236  *     // check PICTL first because if we are in suspend mode then the P0
00237  *     // interrupt settings may be different from what you set them to be.
00238  *
00239  *     if (P0IFG & 0x80)  // Check USB_RESUME bit.
00240  *     {
00241  *         usbSuspendMode = 0;   // Causes usbSleep to exit sleep mode.
00242  *     }
00243  *     P0IFG = 0;   // Clear the flags so this interrupt doesn't run again.
00244  *     P0IF = 0;
00245  * }
00246  *
00247  * If your P0 ISR does NOT clear usbSuspendMode, then your device will not be
00248  * able to wake up out of suspend mode because when USB activity happens, your
00249  * ISR will (if it's well-written) clear P0IFG so usbSleep won't be able to
00250  * see that the P0IFG.USB_RESUME bit has been set, and it will keep on sleeping.
00251  */
00252 extern volatile BIT usbSuspendMode;
