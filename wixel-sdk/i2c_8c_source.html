00091  *  SDA goes high while SCL is high

00131 /* Write a bit to the I2C bus
00132  * It is assumed that SCL is low when this function starts.
00133  * SDA is set to the appropriate bit value while SCL is low, there is a
00134  * delay for half of the clock period while SDA stablizes, then SCL
00135  * is allowed to go high for the second half of the clock period, which
00136  * indicates the on SDA is valid.  This function drives SCL low again
00137  * before it returns.
00138  */

00140 {
00141     if (b)
00142     {
00143         i2cReadSda(); // let SDA go high
00144     }
00145     else
00146     {
00147         i2cClearSda(); // drive SDA low
00148     }
00149     delayMicroseconds(halfPeriodUs);
00150 
00151     // handle clock stretching
00152     i2cWaitForHighScl(timeout);
00153     if (internalTimeoutOccurred) return;
00154 
00155     // SCL is now high, data is valid

00160 /* Read a bit to the I2C bus
00161  * It is assumed that SCL is low when this function starts.
00162  * The master tristates SDA so the slave transmitter can control the state
00163  * and delays for half of the clock period (or longer if the slave is holding
00164  * SCL low).  It then lets SCL go high, records the state of the SDA line,
00165  * and delays for the second half of the clock period. This function drives SCL
00166  * low again before it returns. Return value is not meaningful if timeout
00167  * occurs.
00168  */
00169 BIT i2cReadBit(void)
00170 {
00171     BIT b;
00172 
00173     i2cReadSda(); // let slave transmitter control state of SDA line
00174     delayMicroseconds(halfPeriodUs);
00175 
00176     // handle clock stretching
00177     i2cWaitForHighScl(timeout);
00178     if (internalTimeoutOccurred) return 0;
00179 
00180     // SCL is now high, data is valid
00181     b = i2cReadSda(); // store state of SDA line now that SCL is high
00182     delayMicroseconds(halfPeriodUs);
00183     i2cClearScl(); // drive SCL low
00184     return b;
00185 }
