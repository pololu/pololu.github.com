00373 void usbComService(void)
00374 {
00375     usbPoll();
00376 
00377     // Start bootloader if necessary.
00378     if (startBootloaderSoon && (uint8)(getMs() - startBootloaderRequestTime) > 70)
00379     {
00380         // It has been 50 ms since the user requested that we start the bootloader, so
00381         // start it.
00382 
00383         // The reason we don't start the bootloader right away when we get the request is
00384         // because we want to have time to finish the status phase of the control transfer
00385         // so the host knows the request was processed correctly.  Also, the Windows
00386         // usbser.sys sends several requests for about 7 ms after the SET_LINE_CODING
00387         // request before the operation (SetCommState) finally succeeds.
00388         boardStartBootloader();
00389     }
00390 
00391     if (usbDeviceState != USB_STATE_CONFIGURED)
00392     {
00393         // We have not reached the Configured state yet, so we should not be touching the non-zero endpoints.
00394         return;
00395     }
00396 
00397     // Send a packet now if there is data loaded in the FIFO waiting to be sent OR
00398     //
00399     // Typical USB systems wait for a short or empty packet before forwarding the data
00400     // up to the software that requested it, so this is necessary.  However, we only transmit
00401     // an empty packet if there are no packets currently loaded in the FIFO.
00402     USBINDEX = CDC_DATA_ENDPOINT;
00403     if (inFifoBytesLoaded || ( sendEmptyPacketSoon && !(USBCSIL & USBCSIL_PKT_PRESENT) ) )
00404     {
00405         sendPacketNow();
00406     }
00407 
00408     // Notify the computer of the current serial state if necessary.
00409     USBINDEX = CDC_NOTIFICATION_ENDPOINT;
00410     if (usbComSerialState != lastReportedSerialState && !(USBCSIL & USBCSIL_INPKT_RDY))

00413         // AND we are ready to send it to the USB host, so send it.
