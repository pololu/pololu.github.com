00090      *                     TX  = P0_3
00091      *                     RX  = P0_2
00092      */

00094     /* USART1 UART Alt. 2:
00095      *                     TX  = P1_6
00096      *                     RX  = P1_7
00097      */
00098 
00099     uartTxBufferMainLoopIndex = 0;
00100     uartTxBufferInterruptIndex = 0;
00101     uartRxBufferMainLoopIndex = 0;
00102     uartRxBufferInterruptIndex = 0;
00103     uartNRxParityErrorOccurred = 0;
00104     uartNRxFramingErrorOccurred = 0;
00105     uartNRxBufferFullOccurred = 0;
00106 
00107     // Note: We do NOT set the mode of the RX pin to "peripheral function"
00108     // because that seems to have no benefits, and is actually bad because
00109     // it disables the internal pull-up resistor.
00110 
00111 #ifdef UART0
00112     P2DIR &= ~0xC0;  // P2DIR.PRIP0 (7:6) = 00 : USART0 takes priority over USART1.
00113     PERCFG &= ~0x01; // PERCFG.U0CFG (0) = 0 (Alt. 1) : USART0 uses alt. location 1.
00114 #else
00115     P2SEL |= 0x40;   // USART1 takes priority over USART0 on Port 1.
00116     PERCFG |= 0x02;  // PERCFG.U1CFG (1) = 1 (Alt. 2) : USART1 uses alt. location 2.
00117 #endif
00118 
00119     UNUCR = 0x82;    // Stops the "current operation" and resets settings to their defaults.
00120     UNCSR |= 0xc0;   // Enable UART mode and enable receiver.  TODO: change '|=' to '='
00121 
00122     // Set the mode of the TX pin to "peripheral function".  This must be done AFTER
00123     // enabling the UART, or else we get a tiny glitch on the TX line.
00124 #ifdef UART0
00125     P0SEL |= (1<<3); // P0SEL.SELP0_3 = 1
00126 #else
00127     P1SEL |= (1<<6); // P1SEL.SELP1_6 = 1
00128 #endif
00129 
00130     // Below, we set the priority of the RX and TX interrupts to be 1 (second lowest priority).
00131     // They need to be higher than the RF interrupt because that one could take a long time.
00132     // The UART0 interrupts are grouped with the T2 interrupt, so its priority also gets set.
00133     // The UART1 interrupts are grouped with the T3 interrupts, so its priority also gets set.
00134     IP0 |= (1<<INTERRUPT_PRIORITY_GROUP);
00135     IP1 &= ~(1<<INTERRUPT_PRIORITY_GROUP);
00136 
00137     UTXNIF = 1; // Set TX flag so the interrupt fires when we enable it for the first time.
00138     URXNIF = 0; // Clear RX flag.
00139     URXNIE = 1; // Enable Rx interrupt.
00140     EA = 1;     // Enable interrupts in general.

00144 {
00145     uint32 baudMPlus256;
00146     uint8 baudE = 0;
00147 
00148     // actual max baud rate is 1500000 (F/16) - we are limited by the way we calculate the exponent and mantissa
00149     if (baud < 23 || baud > 495782)
00150         return;
00151 
00152     // calculate baud rate - see datasheet 12.14.3
00153     // this is derived from (baudM + 256) = baud * 2^28 / 24000000
00154     baudMPlus256 = (baud * 11) + (baud * 8663 / 46875);
00155 
00156     // get baudMPlus256 into the range 256-511 (so BAUD_M is in the range 0-255)
00157     while (baudMPlus256 > 0x1ff)
00158     {
00159         baudE++;
00160         baudMPlus256 /= 2;
00161     }
00162     UNGCR = baudE; // UNGCR.BAUD_E (4:0)
00163     UNBAUD = baudMPlus256; // UNBAUD.BAUD_M (7:0) - only the lowest 8 bits of baudMPlus256 are used, so this is effectively baudMPlus256 - 256
00164 }
