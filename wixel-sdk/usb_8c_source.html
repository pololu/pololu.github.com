00286 // usbStandardDeviceRequestHandler(): Implementation of USB2.0 Section 9.4, Standard Device Requests.
00287 // This function gets called whenever we receive a SETUP packet on endpoint zero with the requestType
00288 // field set to STANDARD.  This function reads the SETUP packet and uses that to set the control
00289 // transfer state variables with all the information needed to respond to the request.
00290 // Assumption: controlTransferState is CONTROL_TRANSFER_STATE_NONE when this function is called.
00291 static void usbStandardDeviceRequestHandler()
00292 {
00293     // Prepare a convenient two-byte buffer for sending 1 or 2 byte responses.
00294     static XDATA uint8 response[2];
00295     response[0] = 0;
00296     response[1] = 0;
00297 
00298     // Now we decide how to handle the new setup packet.  There are several possibilities:
00299     // * Invalid: The SETUP packet had a problem with it, or we don't support the feature,
00300     //     so we need to STALL the next transaction to indicate an request error to the host.
00301     // * Control Read:  We must send some data to the computer, so we need to decide
00302     //     where the data is coming from (address, plus RAM/ROM selection)
00303     // * Control Write with no data phase: We need to prepare for the status phase, where
00304     //     our device must send a zero-length EP0 IN packet to indicate success.
00305     // * Control Write with data phase: The computer will send data to us, and we need to
00306     //     decide where in RAM to put it.  (No standard device requests use this type,
00307     //     so ignore this case.)
00308 
00309     switch(usbSetupPacket.bRequest)
00310     {
00311         case USB_REQUEST_GET_DESCRIPTOR: // USB Spec 9.4.3 Get Descriptor
00312         {
00313             switch(usbSetupPacket.wValue >> 8)
00314             {
00315                 case USB_DESCRIPTOR_TYPE_DEVICE:
00316                 {
00317                     controlTransferPointer = (uint8 XDATA *)&usbDeviceDescriptor;
00318                     controlTransferBytesLeft = sizeof(USB_DESCRIPTOR_DEVICE);
00319                     break;
00320                 }
00321                 case USB_DESCRIPTOR_TYPE_CONFIGURATION:
00322                 {
00323                     if ((usbSetupPacket.wValue & 0xFF) != 0)
00324                     {
00325                         // Invalid configuration index.
00326                         return;
00327                     }
00328 
00329                     // The configuration descriptor has an application-dependent size, which
00330                     // we determine by reading the 3rd and 4th byte.
00331                     controlTransferPointer = (uint8 XDATA *)usbConfigurationDescriptor;
00332                     controlTransferBytesLeft = *(uint16 *)&usbConfigurationDescriptor[2];
00333                     break;
00334                 }
00335                 case USB_DESCRIPTOR_TYPE_STRING:
00336                 {
00337                     if ((usbSetupPacket.wValue & 0xFF) >= usbStringDescriptorCount)
00338                     {
00339                         // Invalid string index.
00340                         return;
00341                     }
00342 
00343                     controlTransferPointer = (uint8 XDATA *)usbStringDescriptors[usbSetupPacket.wValue & 0xFF];
00344                     controlTransferBytesLeft = controlTransferPointer[0];
00345                     break;
00346                 }
00347                 default:
00348                 {
00349                     // see if the class recognizes the descriptor type; it should call usbControlRead if it does
00350                     usbCallbackClassDescriptorHandler();
00351 
00352                     if (controlTransferState == CONTROL_TRANSFER_STATE_NONE)
00353                     {
00354                         // unknown type of descriptor
00355                         return;
00356                     }
00357                     break;
00358                 }
00359             }
00360 
00361             // Modify the count so that we don't send more data than the host requested.
00362             // We MUST use the local variable wLength instead of usbSetupPacket.wLength because
00363             // USB_SETUP_PACKET may have been over-written by the serial number handler.
00364             if(controlTransferBytesLeft > usbSetupPacket.wLength)
00365             {
00366                 controlTransferBytesLeft = usbSetupPacket.wLength;
00367             }
00368 
00369             controlTransferState = CONTROL_TRANSFER_STATE_READ;
00370             return;
00371         }
00372         case USB_REQUEST_SET_ADDRESS: // USB Spec, 9.4.6 Set Address
00373         {
00374             // Get ready to set the address when the status phase is complete.
00375             // We always set the most siginificant bit, because .device_address might be 0
00376             // and that is a valid request, meaning we should revert to address 0.
00377             //pendingDeviceAddress = (usbSetupPacket.wValue & 0xFF) | 0x80;
00378 
00379             USBADDR = (uint8)usbSetupPacket.wValue;
00380             usbDeviceState = ((uint8)usbSetupPacket.wValue) ? USB_STATE_ADDRESS : USB_STATE_DEFAULT;
00381 
00382             // Get ready to provide a handshake.
00383             usbControlAcknowledge();
00384             return;
00385         }
00386         case USB_REQUEST_SET_CONFIGURATION: // USB Spec, 9.4.7 Set Configuration
00387         {
00388             // Assumption: there is only one configuration and its value is 1.
00389             switch(usbSetupPacket.wValue)
00390             {
00391                 case 0:
00392                 {
00393                     // We have been deconfigured.
00394 
00395                     // TODO: Add resetNonzeroEndpoints() and call it here.
00396 
00397                     if (usbDeviceState > USB_STATE_ADDRESS)
00398                     {
00399                         usbDeviceState = USB_STATE_ADDRESS;
00400                     }
00401                     break;
00402                 }
00403                 case 1:
00404                 {
00405                     // The device has been configured.  This is normal operating
00406                     // state of a USB device.  We can now start using non-zero
00407                     // endpoints.
00408                     usbDeviceState = USB_STATE_CONFIGURED;
00409                     usbCallbackInitEndpoints();
00410                     break;
00411                 }
00412                 default:
00413                 {
00414                     // Invalid configuration value, so STALL.
00415                     return;
00416                 }
00417             }
00418 
00419             // Get ready to provide a handshake.
00420             usbControlAcknowledge();
00421             return;
00422         }
00423         case USB_REQUEST_GET_CONFIGURATION: // USB Spec 9.4.2 Get Configuration
00424         {
00425             // Assumption: there is only one configuration and its value is 1.
00426             response[0] = (usbDeviceState == USB_STATE_CONFIGURED) ? 1 : 0;
00427             usbControlRead(1, response);
00428             return;
00429         }
00430         case USB_REQUEST_GET_INTERFACE: // USB Spec 9.4.4 Get Interface
00431         {
00432             // Assumption: the "alternate setting number" of each interface
00433             //   is zero and there are no alternate settings.
00434             // Assumption: interface numbers go from 0 to
00435             //   config->interface_count-1, with no gaps.
00436 
00437             if (usbDeviceState < USB_STATE_CONFIGURED)
00438             {
00439                 // Invalid request because we have not reached the configured state.
00440                 return;
00441             }
00442 
00443             if (usbSetupPacket.wIndex >= ((USB_DESCRIPTOR_CONFIGURATION *)&usbConfigurationDescriptor)->bNumInterfaces)
00444             {
00445                 // Invalid index: there is no such interface.
00446                 return;
00447             }
00448 
00449             // Send a single-byte response of "0".
00450             // Assumption: response[0] == 0
00451             usbControlRead(1, response);
00452             return;
00453         }
00454         case USB_REQUEST_GET_STATUS: // USB Spec 9.4.5 Get Status
00455         {
00456             switch(usbSetupPacket.recipient)
00457             {
00458                 case USB_RECIPIENT_DEVICE:
00459                 {
00460                     // See USB Spec Table 9-4.
00461                     response[0] = vinPowerPresent() ? 1 : 0;
00462                     // Assumption: response[1] == 0
00463                     usbControlRead(2, response);
00464                     return;
00465                 }
00466                 case USB_RECIPIENT_INTERFACE:
00467                 {
00468                     if (usbDeviceState < USB_STATE_CONFIGURED && usbSetupPacket.wIndex != 0)
00469                     {
00470                         // It is invalid to ask about interfaces other than 0 before the
00471                         // configured state.
00472                         return;
00473                     }
00474 
00475                     if (usbSetupPacket.wIndex >= ((USB_DESCRIPTOR_CONFIGURATION *)&usbConfigurationDescriptor)->bNumInterfaces)
00476                     {
00477                         // Invalid index: there is no such interface.
00478                         return;
00479                     }
00480 
00481                     // Send a 2-byte response of 0,0 (all of the bits are reserved)
00482                     // Assumption: response[0] == 0 and response[1] == 0
00483                     usbControlRead(2, response);
00484                     return;
00485                 }
00486                 case USB_RECIPIENT_ENDPOINT:
00487                 {
00488                     if ((usbSetupPacket.wValue & 15) == 0)
00489                     {
00490                         // We don't support the halt feature on Endpoint 0
00491                         // (the USB Spec does not require or recommend it).
00492                         return;
00493                     }
00494 
00495                     if (usbDeviceState < USB_STATE_CONFIGURED)
00496                     {
00497                         // It is invalid to ask about non-zero endpoints before
00498                         // the configured state.
00499                         return;
00500                     }
00501 
00502                     // Assumption: We don't have a USB halt feature, i.e. we
00503                     // don't stall on non-zero endpoints.
00504 
00505                     // Send a 2-byte response of 0,0.
00506                     // Assumption: response[0] == 0 and response[1] == 0
00507                     usbControlRead(2, response);
00508                     return;
00509                 }
00510             }
00511             return;
00512         }
00513

00515         // to be USB compliant.  We didn't use them yet on any of our

00517         // know.  We pay lip service to them here just in case they are
00518         // needed by some future driver.
00519         case USB_REQUEST_SET_FEATURE:
00520         case USB_REQUEST_CLEAR_FEATURE:
00521         {
00522             // Acknowledge the request but don't do anything.
00523             usbControlAcknowledge();
00524             return;
00525         }
00526         case USB_REQUEST_SYNCH_FRAME:
00527         {
00528             // Send a two-byte response of 0,0.
00529             usbControlRead(2, response);

00531         }

00540 // Sleeps until we receive USB resume signaling.
00541 // This uses PM1.  ( PM2 and PM3 are not usable because they will reset the USB module. )
00542 // NOTE: For some reason, USB suspend does not work if you plug your device into a computer
00543 // that is already sleeping.  If you do that, the device will remain awake with
00544 // usbDeviceState == USB_STATE_POWERED and it will draw more power than it should from USB.
00545 // TODO: figure out how to wake up when self power is connected.  Probably we should use the
00546 // sleep timer to wake up regularly and check (that's going to be easier than using a P2
00547 // interrupt I think).
00548 // TODO: make sure suspend mode doesn't interfere with the radio libraries.  We should
00549 // probably make a simple power-event registration thing using function pointers,
00550 // (either an array, or a linked list where the memory is contributed by the modules using it).
00551 // When going to sleep, we would call these functions in the order they were added;
00552 // when waking up, we would call them in the opposite order.  Look at how we handle power
00553 // on the jrk, maestros, and simple motor controller to see if this pattern works.
00554 void usbSleep()
00555 {
00556     uint8 savedPICTL = PICTL;
00557     BIT savedP0IE = P0IE;
00558 
00559     // The USB resume interrupt is mapped to the non-existent pin, P0_7.
00560 
00561     P0IE = 0;         // Disable the P0 interrupt while we are reconfiguring it (maybe not necessary).
00562     PICTL |= (1<<4);  // PICTL.P0IENH = 1  Enable the Port 0 interrupts for inputs 4-7 (USB_RESUME is #7).
00563     PICTL &= ~(1<<0); // PICTL.P0ICON = 0  Detect rising edges (this is required for waking up).
00564 
00565     do
00566     {
00567         // Clear the P0 interrupt flag that might prevent us from sleeping.
00568         P0IFG = 0;   // Clear Port 0 module interrupt flags.
00569         P0IF = 0;    // Clear Port 0 CPU interrupt flag (IRCON.P0IF = 0).
00570 
00571         P0IE = 1;    // Enable the Port 0 interrupt (IEN1.P0IE = 1) so we can wake up.
00572 
00573         // Put the device to sleep by following the recommended pseudo code in the datasheet section 12.1.3:
00574         SLEEP = (SLEEP & ~3) | 1;    // SLEEP.MODE = 1 : Selects Power Mode 1 (PM1).
00575         __asm nop __endasm; __asm nop __endasm; __asm nop __endasm;
00576         if (SLEEP & 3)
00577         {
00578             P1_0 = 1;
00579             PCON |= 1;    // PCON.IDLE = 1 : Actually go to sleep.
00580             P1_0 = 0;
00581         }
00582 
00583         // Disable the Port 0 interrupt.  If we don't do this, and there is no ISR
00584         // (just a reti), then the non-existent ISR could run many times while we
00585         // are awake, slowing down this loop.
00586         P0IE = 0; // (IEN1.P0IE = 1)
00587 
00588         // Check to see if the USB_RESUME bit is set.  If it is set, then there was
00589         // activity detected on the USB and it is time to wake up.
00590         // NOTE: The P0INT ISR might also set usbSuspendMode to 0 if the user defines
00591         // that ISR.  See the comment about P0INT in usb.h for more info.
00592         if (P0IFG & 0x80)
00593         {
00594             usbSuspendMode = 0;
00595         }
00596     }
00597     while(usbSuspendMode && !vinPowerPresent());
00598 
00599     // Wait for the high speed crystal oscillator to become stable again
00600     // because we can't write to USB registers until that happens.
00601     while(!(SLEEP & (1<<6))){};
00602 
00603     // Restore the interrupt registers to their original states.
00604     PICTL = savedPICTL;
00605     P0IE = savedP0IE;
00606 }
