00516 // Sleeps until we receive USB resume signaling.
00517 // This uses PM1.  ( PM2 and PM3 are not usable because they will reset the USB module. )
00518 // NOTE: For some reason, USB suspend does not work if you plug your device in to a computer
00519 // that is already sleeping.  If you do that, the device will remain awake with
00520 // usbDeviceState == USB_STATE_POWERED and it will draw more power than it should from USB.
00521 // TODO: figure out how to wake up when self power is connected.  Probably we should use the
00522 // sleep timer to wake up regularly and check (that's going to be easier than using a P2
00523 // interrupt I think).
00524 // TODO: make sure suspend mode doesn't interfere with the radio library.  We should
00525 // probably make a simple power-event registration thing using function pointers,
00526 // (either an array, or a linked list where the memory is contributed by the modules using it).
00527 // When going to sleep, we would call these functions in the order they were added;
00528 // when waking up, we would call them in the opposite order.  Look at how we handle power
00529 // on the jrk, maestros, and simple motor controller to see if this pattern works.
00530 void usbSleep()
00531 {
00532     uint8 savedPICTL = PICTL;
00533     BIT savedP0IE = P0IE;
00534 
00535     // The USB resume interrupt is mapped to the non-existent pin, P0_7.
00536 
00537     P0IE = 0;         // Disable the P0 interrupt while we are reconfiguring it (maybe not necessary).
00538     PICTL |= (1<<4);  // PICTL.P0IENH = 1  Enable the Port 0 interrupts for inputs 4-7 (USB_RESUME is #7).
00539     PICTL &= ~(1<<0); // PICTL.P0ICON = 0  Detect rising edges (this is required for waking up).
00540 
00541     do
00542     {
00543         // Clear the P0 interrupt flag that might prevent us from sleeping.
00544         P0IFG = 0;   // Clear Port 0 module interrupt flags.
00545         P0IF = 0;    // Clear Port 0 CPU interrupt flag (IRCON.P0IF = 0).
00546 
00547         P0IE = 1;    // Enable the Port 0 interrupt (IEN1.P0IE = 1) so we can wake up.
00548 
00549         // Put the device to sleep by following the recommended pseudo code in the datasheet section 12.1.3:
00550         SLEEP = (SLEEP & ~3) | 1;    // SLEEP.MODE = 1 : Selects Power Mode 1 (PM1).
00551         __asm nop __endasm; __asm nop __endasm; __asm nop __endasm;
00552         if (SLEEP & 3)
00553         {
00554             P1_0 = 1;
00555             PCON |= 1;    // PCON.IDLE = 1 : Actually go to sleep.
00556             P1_0 = 0;
00557         }
00558 
00559         // Disable the Port 0 interrupt.  If we don't do this, and there is no ISR
00560         // (just a reti), then the non-existent ISR could run many times while we
00561         // are awake, slowing down this loop.
00562         P0IE = 0; // (IEN1.P0IE = 1)
00563 
00564         // Check to see if the USB_RESUME bit is set.  If it is set, then there was
00565         // activity detected on the USB and it is time to wake up.
00566         // NOTE: The P0INT ISR might also set usbSuspendMode to 0 if the user defines
00567         // that ISR.  See the comment about P0INT in usb.h for more info.
00568         if (P0IFG & 0x80)
00569         {
00570             usbSuspendMode = 0;
00571         }
00572     }
00573     while(usbSuspendMode && !vinPowerPresent());
00574 
00575     // Wait for the high speed crystal oscillator to become stable again
00576     // because we can't write to USB registers until that happens.
00577     while(!(SLEEP & (1<<6))){};
00578 
00579     // Restore the interrupt registers to their original states.
00580     PICTL = savedPICTL;
00581     P0IE = savedP0IE;
00582 }
