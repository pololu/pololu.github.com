00001 /*  NOTE: Calibration of the frequency synthesizer and other RF hardware takes about 800 us and
00002  *  must be done regularly.  There are several options for when to do the calibration and not.
00003  *  We configured the radio to automatically calibrate whenever going from the IDLE state to TX
00004  *  or RX (MCSM0.FS_AUTOCAL = 01).  The radio will go into the idle state whenever the is an RX
00005  *  timeout.  However, to enable a quick turnaround between TX and RX, we configured the radio to
00006  *  automatically go into the FSTXON mode after it is done with RX or TX mode.  FSTXON means that
00007  *  the frequency synthesizer is on and the radio is ready to go into RX or TX mode quickly
00008  *  (but it goes to TX mode faster).
00009  *
00010  *  So basically we are depending on the RX timeout feature to schedule our calibrations for us,
00011  *  instead of doing it manually.  This should work, but it will probaby hinder our ability to
00012  *  quickly recover from lost packets (the RX timeout event is what happens when a packet is lost).
00013  *  An easy alternative would be to only calibrate after every 10th RX timeout or something like
00014  *  that.
00015  */
00016 
00017 /*  The definition of the maximum packet size (and the code that sets the PKTLEN register) is not
00018  *  in this layer.  That is up to the higher-level code (radio_link.c) to decide.   When this
00019  *  layer needs to know the packet size (for setting up the DMA), it reads it from PKTLEN.  This
00020  *  makes the code a little less efficient because it takes longer to access an XDATA register
00021  *  than a hardcoded constant, but makes this MAC layer much more reusable.
00022  */

00057 ISR(RF, 0)
00058 {
00059     S1CON = 0; // Clear the general RFIF interrupt registers
00060 
00061     if (RFIF & 0x10) // Check IRQ_DONE
00062     {
00063         if (radioMacState == RADIO_MAC_STATE_TX)
00064         {
00065             // We just sent a packet.
00066             radioMacEvent(RADIO_MAC_EVENT_TX);
00067         }
00068         else if (radioMacState == RADIO_MAC_STATE_RX)
00069         {
00070             // We just received a packet, but it might have an invalid CRC or be irrelevant
00071             // for other reasons.
00072             radioMacEvent(RADIO_MAC_EVENT_RX);
00073         }
00074     }
00075 
00076     if (RFIF & 0x20)  // Check IRQ_TIMEOUT
00077     {
00078         // We were listening for packets but we didn't receive anything
00079         // and the timeout period expired.
00080         radioMacEvent(RADIO_MAC_EVENT_RX_TIMEOUT);
00081     }
00082 
00083     if (strobe)
00084     {
00085         // Some other code has set the strobe bit, which means he wants the radioMacEventHandler to
00086         // run soon, typically because new data is available for it to send.
00087         if (radioMacState == RADIO_MAC_STATE_TX)
00088         {
00089             // We are currently transmitting, so we will wait for the end of that packet.
00090             // Then, we will issue a RADIO_MAC_EVENT_TX.
00091             return;
00092         }
00093 
00094         if (radioMacState == RADIO_MAC_STATE_RX && MARCSTATE == 0x0D)
00095         {
00096             if (PKTSTATUS & (1<<3))  // Check SFD bit (Start of Frame Delimiter)
00097             {
00098                 // We are currently receiving a packet, so we will wait for the end of that
00099                 // packet and then issue a RADIO_MAC_EVENT_RX.
00100                 // ASSUMPTION: There is no automatic address filtering, and packets with
00101                 // bad CRCs still result in a RAIDO_MAC_EVENT_RX.
00102                 return;
00103             }
00104             if ((MCSM2&7) != 7 && WOREVT1 < MAX_LATENCY_OF_STROBE)
00105             {
00106                 // We are currently listening for a packet and the timeout is going to happen
00107                 // soon (within 10 ms if MSCM2==1 or 20 ms if MCSMS2==0), so we will not actually issue a RADIO_MAC_EVENT_STROBE
00108                 // right now.
00109                 return;
00110             }
00111         }
00112 
00113 
00114         /* The code below is necessary because we found that if the radio is in the
00115            process of calibrating itself to go into RX mode, it won't respond
00116            correctly to an STX strobe (it goes into RX mode instead of TX).
00117            We only need to worry about that here, and not in the other events,
00118            because those other events only happen at times when the radio should not
00119            be in the middle of calibrating itself. */
00120         if (MARCSTATE != 0x0D)
00121         {
00122             RFST = SIDLE;
00123         }
00124 
00125         // We are not currently transmitting and nothing is being received at the
00126         // moment, so we should stop and issue a RADIO_MAC_EVENT_STROBE now.
00127         radioMacEvent(RADIO_MAC_EVENT_STROBE);
00128     }
00129 
00130     if (RFIF & 0x80)   // Check IRQ_TXUNF
00131     {
00132         // We were not sending data to the radio fast enough, so there was a
00133         // TX underflow.  This should not happen because we use DMA to send
00134         // the data.  Report it as an error.
00135         radioTxUnderflowOccurred = 1;
00136         RFIF = ~0x80;
00137     }
00138 
00139     if (RFIF & 0x40)   // Check IRQ_RXOVF
00140     {
00141 
00142         // We were not reading data from the radio fast enough, so there was
00143         // a RX overflow.  This should not happen.  Report it as an error.
00144         radioRxOverflowOccurred = 1;
00145         RFIF = ~0x40;
00146 
00147         // The radio module is probably now in the RX_OVERFLOW state where it can not
00148         // receive packets.  The way to get out of this state is:
00149         //RFST = SIDLE;  (and you probably should put NOPs here, though in my tests it was not necessary)
00150         //RFST = SRX;
00151         // We never expect an RX overflow to happen though, so we won't do that.
00152     }
00153 }

00155 void radioMacEvent(uint8 event)
00156 {
00158     /* This is necessary because David has observed that sometimes (maybe every
00159      * time?) when a packet with a bad CRC is received, the radio stays in RX
00160      * instead of going to FSTXON mode the way it should (RXOFF_MODE=01).
00161      * If we allow the radio to stay in RX mode then an RX overflow error could
00162      * happen later after we disarm the DMA channel. */
00163     if (MARCSTATE != 0x12 && MARCSTATE != 0x01 && MARCSTATE != 0x00 && MARCSTATE != 0x15)
00164     {
00165         // Fix the bad state by telling the radio to go to the SFSTXON state.
00166         RFST = SFSTXON;
00167     }
00168 
00170     DMAARM = 0x80 | (1<<DMA_CHANNEL_RADIO); // Abort any ongoing radio DMA transfer.
00171     DMAIRQ &= ~(1<<DMA_CHANNEL_RADIO);      // Clear any pending radio DMA interrupt
00172 
00174     radioMacState = RADIO_MAC_STATE_RX;    // Default next state: RX
00175     MCSM2 = 0x07;                          // Default next timeout: infinite.
00176     radioMacEventHandler(event);
00177 
00179     // We want to do it before restarting the radio (to avoid accidentally missing
00180     // an event) but we want to do it as long as possible AFTER turning off the
00181     // radio.
00182     RFIF = ~0x30;  // Clear IRQ_DONE and IRQ_TIMEOUT if they are set.
00183 
00185     switch(radioMacState)
00186     {
00187     case RADIO_MAC_STATE_RX:
00188         DMAARM |= (1<<DMA_CHANNEL_RADIO);   // Arm DMA channel.
00189         RFST = SRX;                         // Switch radio to RX.
00190         break;
00191     case RADIO_MAC_STATE_TX:
00192         DMAARM |= (1<<DMA_CHANNEL_RADIO);   // Arm DMA channel.
00193         RFST = STX;                         // Switch radio to TX.
00194         break;
00195     }
00196 
00197     // Clear the strobe bit because we just ran the radioMacEventHandler.
00198     strobe = 0;
00199 }

00226 void radioMacRx(uint8 XDATA * packet, uint8 timeout)
00227 {
00228     if (timeout)
00229     {
00230         MCSM2 = 0x00;   // RX_TIME = 0.  Helps determine the units of the RX timeout period.
00231         WORCTRL = 0;    // WOR_RES = 0.  Helps determine the units of the RX timeout period.

00233         WOREVT0 = 0;

00235     else
00236     {
00237         MCSM2 = 0x07;  // RX_TIME = 7: No timeout.

00239 
00240     dmaConfig.radio.SRCADDRH = XDATA_SFR_ADDRESS(RFD) >> 8;
00241     dmaConfig.radio.SRCADDRL = XDATA_SFR_ADDRESS(RFD);
00242     dmaConfig.radio.DESTADDRH = (unsigned int)packet >> 8;
00243     dmaConfig.radio.DESTADDRL = (unsigned int)packet;
00244     dmaConfig.radio.LENL = 1 + PKTLEN + 2;
00245     dmaConfig.radio.VLEN_LENH = 0b10000000; // Transfer length is FirstByte+3
00246     // Assumption: DC6 is set correctly
00247     dmaConfig.radio.DC7 = 0x10; // SRCINC = 0, DESTINC = 1, IRQMASK = 0, M8 = 0, PRIORITY = 0
00248 
00249     radioMacState = RADIO_MAC_STATE_RX;
