00001 /* radio_link.c:
00002  *  This layer uses radio_mac.c in order to provide reliable ordered delivery and reception of
00003  *  a series of data packets between this device and another.  This is the layer that takes care
00004  *  of Ping/ACK/NAK packets, and handles the details of timing (however, the timing details
00005  *  are dependent on the performance/configuration of the MAC layer, so if the MAC layer
00006  *  changes then the timing details here may have to change.)
00007  *
00008  *  This layer defines the RF packet memory buffers used, and controls access to those buffers.
00009  *
00010  *  This layer also takes care of presence detection by sending out regular Ping packets even
00011  *  if there is no data packet to transmit.
00012  *
00013  *  This is the layer that restricts us to only talk to one device.  (If you wanted to make a
00014  *  network with more than 2 devices, you would have to replace this layer with something more
00015  *  complicated that has addressing.)
00016  *
00017  *  Similarly, this layer also restricts us to only having one logical data pipe.  If you wanted
00018  *  to send some extra data that doesn't get NAKed, or gets NAKed at different times then the
00019  *  regular data, you would need to replace this layer with something more complicated that
00020  *  keeps track of different streams and schedules them in a reasonable way.
00021  *
00022  *  There is a distinction here between RF packets and data packets:  an RF packet is what gets
00023  *  transmitted by the radio.  A data packet is a piece of data that needs to be sent to the
00024  *  other device, and it might correspond to several RF packets because there are retries and
00025  *  ACKs.
00026  *
00027  *  This layer does not correspond cleanly to any of the layers in the OSI Model.
00028  *  It combines portions of the Data Link Layer (#2), Network Layer (#3), and
00029  *  Transport Layer (#4).
00030  */

00060 /*  rxPackets:
00061  *  We need to be prepared at all times to receive a full packet from the other party,
00062  *  even if all we can do is NAK it.  Therefore, we need (at least) THREE buffers, so
00063  *  that two can be owned by the main loop while another is owned by the ISR and ready
00064  *  to receive the next packet.
00065  *
00066  *  If a packet is received and the main loop still owns the other two buffers,
00067  *  we respond with a NAK to the other device.
00068  *
00069  *  Ownership of the RX packet buffers is determined from radioLinkRxMainLoopIndex and radioLinkRxInterruptIndex.
00070  *  The main loop owns all the buffers from radioLinkRxMainLoopIndex to radioLinkRxInterruptIndex-1 inclusive.
00071  *  If the two indices are equal, then the main loop owns nothing.  Here are three examples:
00072  *
00073  *  radioLinkRxMainLoopIndex | radioLinkRxInterruptIndex | Buffers owned by main loop.
00074  *                0 |                0 | None
00075  *                0 |                1 | rxBuffer[0]
00076  *                0 |                2 | rxBuffer[0 and 1]
00077  */
00078 #define RX_PACKET_COUNT  3
00079 static volatile uint8 XDATA radioLinkRxPacket[RX_PACKET_COUNT][1 + RADIO_MAX_PACKET_SIZE + 2];  // The first byte is the length, 2nd byte is link header.
00080 volatile uint8 DATA radioLinkRxMainLoopIndex = 0;   // The index of the next rxBuffer to read from the main loop.
00081 volatile uint8 DATA radioLinkRxInterruptIndex = 0;  // The index of the next rxBuffer to write to when a packet comes from the radio.
00082

00087 volatile uint8 DATA radioLinkTxInterruptIndex = 0;  // The index of the current txPacket we are trying to send on the radio.

00097 /* SEQUENCING VARIABLES *******************************************************/
00098 /* Each data packet we transmit contains a bit that is either 0 or 1 called the
00099    sequence bit.  This bit changes every time we send a different data packet,
00100    so it allows the receiver to tell if we transmitted the same packet twice
00101    (which is what happens when the receiver's ACK packet was lost).  */
00102 
00103 // rxSequenceBit is the value of the sequence bit for the LAST packet we received.
00104 // If we receive a packet whose sequence bit has the same value, we will ACK it
00105 // but not pass the data to the main loop.  This variable is only used in the ISR.
00106 static volatile BIT rxSequenceBit;
00107 
00108 // txSequenceBit is the value of the sequence bit for the NEXT packet we will
00109 // send.
00110 static volatile BIT txSequenceBit;

00127 // Returns a random delay in units of 0.922 ms (the same units of radioMacRx).
00128 // This is used to decide how long to wait before retransmitting.
00129 // This is used to decide when to next transmit a queued data packet.
00130 // If we have already tried sending this packet many times, then this function
00131 // will return a much longer delay, in order to avoid overcrowding the airwaves for
00132 // no reason.  This is similar to the idea of exponential backoff used in other
00133 // communications protocols such as Ethernet:
00134 // http://en.wikipedia.org/wiki/Exponential_backoff
00135 static uint8 randomTxDelay()
00136 {
00137     // 200 and 250 were chosen arbitrarily.
00138     return (radioLinkTxCurrentPacketTries > 200 ? 250 : 1) + (randomNumber() & 3);
00139 }
