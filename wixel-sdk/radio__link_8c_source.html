00001 /* radio_link.c:
00002  *  This layer uses radio_mac.c in order to provide reliable ordered delivery and reception of
00003  *  a series of data packets between this device and another.  This is the layer that takes care
00004  *  of Ping/ACK/NAK packets, and handles the details of timing (however, the timing details
00005  *  are dependent on the performance/configuration of the MAC layer, so if the MAC layer
00006  *  changes then the timing details here may have to change.)
00007  *
00008  *  There is a distinction here between RF packets and data packets:  an RF packet is what gets
00009  *  transmitted by the radio.  A data packet is a piece of data that needs to be sent to the
00010  *  other device, and it might correspond to several RF packets because there are retries and
00011  *  ACKs.
00012  */

00036 #define PACKET_TYPE_MASK  (3 << 6) // These are the bits that determine the packet type.
00037 #define PACKET_TYPE_PING  (0 << 6) // If both bits are zero, it is just a Ping packet (with optional data).
00038 #define PACKET_TYPE_NAK   (1 << 6) // A NAK packet (with optional data)
00039 #define PACKET_TYPE_ACK   (2 << 6) // An ACK packet (with optional data)
00040 #define PACKET_TYPE_RESET (3 << 6) // A Reset packet (the next packet transmitted by the sender of this packet will have a sequence number of 0)

00042 /*  rxPackets:
00043  *  We need to be prepared at all times to receive a full packet from the other party,
00044  *  even if all we can do is NAK it.  Therefore, we need (at least) THREE buffers, so
00045  *  that two can be owned by the main loop while another is owned by the ISR and ready
00046  *  to receive the next packet.
00047  *
00048  *  If a packet is received and the main loop still owns the other two buffers,
00049  *  we respond with a NAK to the other device.
00050  *
00051  *  Ownership of the RX packet buffers is determined from radioLinkRxMainLoopIndex and radioLinkRxInterruptIndex.
00052  *  The main loop owns all the buffers from radioLinkRxMainLoopIndex to radioLinkRxInterruptIndex-1 inclusive.
00053  *  If the two indices are equal, then the main loop owns nothing.  Here are three examples:
00054  *
00055  *  radioLinkRxMainLoopIndex | radioLinkRxInterruptIndex | Buffers owned by main loop.
00056  *                0 |                0 | None
00057  *                0 |                1 | rxBuffer[0]
00058  *                0 |                2 | rxBuffer[0 and 1]
00059  */
00060 #define RX_PACKET_COUNT  3
00061 static volatile uint8 XDATA radioLinkRxPacket[RX_PACKET_COUNT][1 + RADIO_MAX_PACKET_SIZE + 2];  // The first byte is the length, 2nd byte is link header.
00062 volatile uint8 DATA radioLinkRxMainLoopIndex = 0;   // The index of the next rxBuffer to read from the main loop.
00063 volatile uint8 DATA radioLinkRxInterruptIndex = 0;  // The index of the next rxBuffer to write to when a packet comes from the radio.
00064

00083 /* SEQUENCING VARIABLES *******************************************************/
00084 /* Each data packet we transmit contains a bit that is either 0 or 1 called the
00085    sequence bit.  This bit changes every time we send a different data packet,
00086    so it allows the receiver to tell if we transmitted the same packet twice
00087    (which is what happens when the receiver's ACK packet was lost).  */
00088 
00089 // rxSequenceBit is the value of the sequence bit for the LAST packet we received.
00090 // If we receive a packet whose sequence bit has the same value, we will ACK it
00091 // but not pass the data to the main loop.  This variable is only used in the ISR.
00092 static volatile BIT rxSequenceBit;
00093 
00094 // txSequenceBit is the value of the sequence bit for the NEXT packet we will

00124 // Returns a random delay in units of 0.922 ms (the same units of radioMacRx).
00125 // This is used to decide how long to wait before retransmitting.
00126 // This is used to decide when to next transmit a queued data packet.
00127 // If we have already tried sending this packet many times, then this function
00128 // will return a much longer delay, in order to avoid overcrowding the airwaves for
00129 // no reason.  This is similar to the idea of exponential backoff used in other
00130 // communications protocols such as Ethernet:
00131 // http://en.wikipedia.org/wiki/Exponential_backoff
00132 static uint8 randomTxDelay()
00133 {
00134     // 200 and 250 were chosen arbitrarily.
00135     return (radioLinkTxCurrentPacketTries > 200 ? 250 : 1) + (randomNumber() & 3);
00136 }
