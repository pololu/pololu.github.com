205  void applySettings()
206  {
207  // Because of power interruption considerations, the register that
208  // contains the MOTEN bit (CR2) must be written first, and whenever we
209  // write to it we should also write to all the other registers.
210 
211  // CR2 is written first, because it contains the MOTEN bit, and there is
212  // a risk that there might be a power interruption to the driver right
213  // before CR2 is written. This could result in the motor being enabled
214  // with incorrect settings. Also, whenever we do write to CR2, we want to
215  // also write the other registers to make sure they are in the correct state.
216  driver.writeReg(CR2, cr2);
217 
218  writeWR();
219  writeCR0();
220  writeCR1();
221  writeCR3();
222  }

337  void setStepMode(uint8_t mode)
338  {
339  // Pick 1/32 micro-step by default.
340  uint8_t esm = 0b000;
341  uint8_t sm = 0b000;
342 
343  // The order of these cases matches the order in Table 12 of the
344  // AMIS-30543 datasheet.
345  switch(mode)
346  {
347  case MicroStep32: sm = 0b000; break;
348  case MicroStep16: sm = 0b001; break;
349  case MicroStep8: sm = 0b010; break;
350  case MicroStep4: sm = 0b011; break;
351  case CompensatedHalf: sm = 0b100; break; /* a.k.a. MicroStep2 */
352  case UncompensatedHalf: sm = 0b101; break;
353  case UncompensatedFull: sm = 0b110; break;
354  case MicroStep128: esm = 0b001; break;
355  case MicroStep64: esm = 0b010; break;
356  case CompensatedFullTwoPhaseOn: esm = 0b011; break; /* a.k.a. MicroStep 1 */
357  case CompensatedFullOnePhaseOn: esm = 0b100; break;
358  }
359 
360  cr0 = (cr0 & ~0b11100000) | (sm << 5);
361  cr3 = (cr3 & ~0b111) | esm;
362  writeCR0();
363  writeCR3();
364  }
