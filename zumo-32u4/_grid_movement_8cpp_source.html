64 // Calibrates the line sensors by turning left and right, then
65 // displays a bar graph of calibrated sensor readings on the LCD.
66 // Returns after the user presses A.
67 static void lineSensorSetup()
68 {

77  // necessary, and so that if there are issues with the gyro
78  // then you will know before actually starting the robot.
79 
80  turnSensorReset();

82  // Turn to the left 90 degrees.
83  motors.setSpeeds(-calibrationSpeed, calibrationSpeed);
84  while((int32_t)turnAngle < turnAngle45 * 2)
85  {
86  lineSensors.calibrate();
87  turnSensorUpdate();
88  }
89 
90  // Turn to the right 90 degrees.
91  motors.setSpeeds(calibrationSpeed, -calibrationSpeed);
92  while((int32_t)turnAngle > -turnAngle45 * 2)
93  {
94  lineSensors.calibrate();
95  turnSensorUpdate();
96  }
97

99  motors.setSpeeds(-calibrationSpeed, calibrationSpeed);
100  while((int32_t)turnAngle < 0)

102  lineSensors.calibrate();
103  turnSensorUpdate();

127 // Turns according to the parameter dir, which should be 'L'
128 // (left), 'R' (right), 'S' (straight), or 'B' (back). We turn
129 // most of the way using the gyro, and then use one of the line
130 // sensors to finish the turn. We use the inner line sensor that
131 // is closer to the target line in order to reduce overshoot.
132 void turn(char dir)
133 {
134  if (dir == 'S')
135  {
136  // Don't do anything!
137  return;
138  }
139 
140  turnSensorReset();
141 
142  uint8_t sensorIndex;
143 
144  switch(dir)
145  {
146  case 'B':
147  // Turn left 125 degrees using the gyro.
148  motors.setSpeeds(-turnSpeed, turnSpeed);
149  while((int32_t)turnAngle < turnAngle45 * 3)
150  {
151  turnSensorUpdate();
152  }
153  sensorIndex = 1;
154  break;
155 
156  case 'L':
157  // Turn left 45 degrees using the gyro.
158  motors.setSpeeds(-turnSpeed, turnSpeed);
159  while((int32_t)turnAngle < turnAngle45)
160  {
161  turnSensorUpdate();
162  }
163  sensorIndex = 1;
164  break;
165 
166  case 'R':
167  // Turn right 45 degrees using the gyro.
168  motors.setSpeeds(turnSpeed, -turnSpeed);
169  while((int32_t)turnAngle > -turnAngle45)
170  {
171  turnSensorUpdate();
172  }
173  sensorIndex = 3;
174  break;
175 
176  default:
177  // This should not happen.
178  return;
179  }
180 
181  // Turn the rest of the way using the line sensors.
182  while(1)
183  {
184  readSensors();
185  if (aboveLine(sensorIndex))
186  {
187  // We found the line again, so the turn is done.
188  break;
189  }
190  }
191 }

193 // This function causes the robot to follow a line segment until
194 // it detects an intersection, a dead end, or a dark spot.
195 void followSegment()
196 {
197  while(1)
198  {
199  // Get the position of the line.
200  uint16_t position = readSensors();
201 
202  // Our "error" is how far we are away from the center of the
203  // line, which corresponds to position 2000.
204  int16_t error = (int16_t)position - 2000;
205 
206  // Compute the difference between the two motor power
207  // settings, leftSpeed - rightSpeed.
208  int16_t speedDifference = error / 4;
209 
210  // Get individual motor speeds. The sign of speedDifference
211  // determines if the robot turns left or right.
212  int16_t leftSpeed = (int16_t)straightSpeed + speedDifference;
213  int16_t rightSpeed = (int16_t)straightSpeed - speedDifference;
214 
215  // Constrain our motor speeds to be between 0 and straightSpeed.
216  leftSpeed = constrain(leftSpeed, 0, (int16_t)straightSpeed);
217  rightSpeed = constrain(rightSpeed, 0, (int16_t)straightSpeed);
218 
219  motors.setSpeeds(leftSpeed, rightSpeed);
220 
221  // We use the inner four sensors (1, 2, 3, and 4) for
222  // determining whether there is a line straight ahead, and the
223  // sensors 0 and 5 for detecting lines going to the left and
224  // right.
225  //
226  // This code could be improved by skipping the checks below
227  // if less than 200 ms has passed since the beginning of this
228  // function. Maze solvers sometimes end up in a bad position
229  // after a turn, and if one of the far sensors is over the
230  // line then it could cause a false intersection detection.
231 
232  if(!aboveLine(0) && !aboveLine(1) && !aboveLine(2) && !aboveLine(3) && !aboveLine(4))
233  {
234  // There is no line visible ahead, and we didn't see any
235  // intersection. Must be a dead end.
236  break;
237  }
238 
239  if(aboveLine(0) || aboveLine(4))
240  {
241  // Found an intersection or a dark spot.
242  break;
243  }
244  }
245 }

256 // This should be called after followSegment to drive to the
257 // center of an intersection. It also uses the line sensors to
258 // detect left, straight, and right exits.
259 void driveToIntersectionCenter(bool * foundLeft, bool * foundStraight, bool * foundRight)
260 {
261  *foundLeft = 0;
262  *foundStraight = 0;
263  *foundRight = 0;
264 
265  // Drive stright forward to get to the center of the
266  // intersection, while simultaneously checking for left and
267  // right exits.
268  //
269  // readSensors() takes approximately 2 ms to run, so we use
270  // it for our loop timing. A more robust approach would be
271  // to use millis() for timing.
272  motors.setSpeeds(straightSpeed, straightSpeed);
273  for(uint16_t i = 0; i < intersectionDelay / 2; i++)
274  {
275  readSensors();
276  if(aboveLine(0))
277  {
278  *foundLeft = 1;
279  }
280  if(aboveLine(4))
281  {
282  *foundRight = 1;
283  }
284  }
285 
286  readSensors();
287 
288  // Check for a straight exit.
289  if(aboveLine(1) || aboveLine(2) || aboveLine(3))
290  {
291  *foundStraight = 1;
292  }
293 }
