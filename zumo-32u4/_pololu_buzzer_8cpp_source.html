116 {
117  DISABLE_TIMER_INTERRUPT();
118 
119 #ifdef __AVR_ATmega32U4__
120  TCCR4A = 0x00; // bits 7 and 6 clear: normal port op., OC4A disconnected
121  // bits 5 and 4 clear: normal port op., OC4B disconnected
122  // bit 3 clear: no force output compare for channel A
123  // bit 2 clear: no force output compare for channel B
124  // bit 1 clear: disable PWM for channel A
125  // bit 0 clear: disable PWM for channel B
126 
127  TCCR4B = 0x04; // bit 7 clear: disable PWM inversion
128  // bit 6 clear: no prescaler reset
129  // bits 5 and 4 clear: dead time prescaler 1
130  // bit 3 clear, 2 set, 1-0 clear: timer clock = CK/8
131 
132  TCCR4C = 0x09; // bits 7 and 6 clear: normal port op., OC4A disconnected
133  // bits 5 and 4 clear: normal port op., OC4B disconnected
134  // bit 3 set, 2 clear: clear OC4D on comp match when upcounting,
135  // set OC4D on comp match when downcounting
136  // bit 1 clear: no force output compare for channel D
137  // bit 0 set: enable PWM for channel 4
138 
139  TCCR4D = 0x01; // bit 7 clear: disable fault protection interrupt
140  // bit 6 clear: disable fault protection mode
141  // bit 5 clear: disable fault protection noise canceler
142  // bit 4 clear: falling edge triggers fault
143  // bit 3 clear: disable fault protection analog comparator
144  // bit 2 clear: fault protection interrupt flag
145  // bit 1 clear, 0 set: select waveform generation mode,
146  // phase- and frequency-correct PWM, TOP = OCR4C,
147  // OCR4D set at BOTTOM, TOV4 flag set at BOTTOM
148 
149  // This sets timer 4 to run in phase- and frequency-correct PWM mode,
150  // where TOP = OCR4C, OCR4D is updated at BOTTOM, TOV1 Flag is set on BOTTOM.
151  // OC4D is cleared on compare match when upcounting, set on compare
152  // match when downcounting; OC4A and OC4B are disconnected.
153 
154  unsigned int top = (F_CPU/16) / 1000; // set TOP for freq = 1 kHz:
155  TC4H = top >> 8; // top 2 bits...
156  OCR4C = top; // and bottom 8 bits
157  TC4H = 0; // 0% duty cycle: top 2 bits...
158  OCR4D = 0; // and bottom 8 bits
159 #else
160  TCCR2A = 0x21; // bits 7 and 6 clear: normal port op., OC4A disconnected
161  // bit 5 set, 4 clear: clear OC2B on comp match when upcounting,
162  // set OC2B on comp match when downcounting
163  // bits 3 and 2: not used
164  // bit 1 clear, 0 set: combine with bit 3 of TCCR2B...
165 
166  TCCR2B = 0x0B; // bit 7 clear: no force output compare for channel A
167  // bit 6 clear: no force output compare for channel B
168  // bits 5 and 4: not used
169  // bit 3 set: combine with bits 1 and 0 of TCCR2A to
170  // select waveform generation mode 5, phase-correct PWM,
171  // TOP = OCR2A, OCR2B set at TOP, TOV2 flag set at BOTTOM
172  // bit 2 clear, 1-0 set: timer clock = clkT2S/32

179  // cycle of the new frequency will be at the old duty cycle, since
180  // the duty cycle (OCR2B) is not updated until TOP.

192 // Set up the timer to play the desired frequency (in Hz or .1 Hz) for the
193 // the desired duration (in ms). Allowed frequencies are 40 Hz to 10 kHz.
194 // volume controls buzzer volume, with 15 being loudest and 0 being quietest.
195 // Note: frequency*duration/1000 must be less than 0xFFFF (65535). This
196 // means that you can't use a max duration of 65535 ms for frequencies
197 // greater than 1 kHz. For example, the max duration you can use for a
198 // frequency of 10 kHz is 6553 ms. If you use a duration longer than this,
199 // you will cause an integer overflow that produces unexpected behavior.
200 void PololuBuzzer::playFrequency(unsigned int freq, unsigned int dur,
201  unsigned char volume)
202 {
203  init(); // initializes the buzzer if necessary
204  buzzerFinished = 0;
205 
206  unsigned int timeout;
207  unsigned char multiplier = 1;
208 
209  if (freq & DIV_BY_10) // if frequency's DIV_BY_10 bit is set
210  { // then the true frequency is freq/10
211  multiplier = 10; // (gives higher resolution for small freqs)
212  freq &= ~DIV_BY_10; // clear DIV_BY_10 bit
213  }
214 
215  unsigned char min = 40 * multiplier;
216  if (freq < min) // min frequency allowed is 40 Hz
217  freq = min;
218  if (multiplier == 1 && freq > 10000)
219  freq = 10000; // max frequency allowed is 10kHz
220 
221 #ifdef __AVR_ATmega32U4__
222  unsigned long top;
223  unsigned char dividerExponent = 0;
224 
225  // calculate necessary clock source and counter top value to get freq
226  top = (unsigned int)(((F_CPU/2 * multiplier) + (freq >> 1))/ freq);
227 
228  while (top > 1023)
229  {
230  dividerExponent++;
231  top = (unsigned int)((((F_CPU/2 >> (dividerExponent)) * multiplier) + (freq >> 1))/ freq);
232  }
233 #else
234  unsigned int top;
235  unsigned char newCS2 = 2; // try prescaler divider of 8 first (minimum necessary for 10 kHz)
236  unsigned int divider = cs2_divider[newCS2];
237 
238  // calculate necessary clock source and counter top value to get freq
239  top = (unsigned int)(((F_CPU/16 * multiplier) + (freq >> 1))/ freq);
240 
241  while (top > 255)
242  {
243  divider = cs2_divider[++newCS2];
244  top = (unsigned int)(((F_CPU/2/divider * multiplier) + (freq >> 1))/ freq);
245  }
246 #endif
247 
248  // set timeout (duration):
249  if (multiplier == 10)
250  freq = (freq + 5) / 10;
251 
252  if (freq == 1000)
253  timeout = dur; // duration for silent notes is exact
254  else
255  timeout = (unsigned int)((long)dur * freq / 1000);
256 
257  if (volume > 15)
258  volume = 15;
259 
260  DISABLE_TIMER_INTERRUPT(); // disable interrupts while writing to registers
261 
262 #ifdef __AVR_ATmega32U4__
263  TCCR4B = (TCCR4B & 0xF0) | (dividerExponent + 1); // select timer 4 clock prescaler: divider = 2^n if CS4 = n+1
264  TC4H = top >> 8; // set timer 4 pwm frequency: top 2 bits...
265  OCR4C = top; // and bottom 8 bits
266  unsigned int width = top >> (16 - volume); // set duty cycle (volume):
267  TC4H = width >> 8; // top 2 bits...
268  OCR4D = width; // and bottom 8 bits
269  buzzerTimeout = timeout; // set buzzer duration
270 
271  TIFR4 |= 0xFF; // clear any pending t4 overflow int.
272 #else
273  TCCR2B = (TCCR2B & 0xF8) | newCS2; // select timer 2 clock prescaler
274  OCR2A = top; // set timer 2 pwm frequency
275  OCR2B = top >> (16 - volume); // set duty cycle (volume)
276  buzzerTimeout = timeout; // set buzzer duration
277 
278  TIFR2 |= 0xFF; // clear any pending t2 overflow int.
279 #endif
280 
281  ENABLE_TIMER_INTERRUPT();
282 }

286 // Determine the frequency for the specified note, then play that note
287 // for the desired duration (in ms). This is done without using floats
288 // and without having to loop. volume controls buzzer volume, with 15 being
289 // loudest and 0 being quietest.
290 // Note: frequency*duration/1000 must be less than 0xFFFF (65535). This
291 // means that you can't use a max duration of 65535 ms for frequencies
292 // greater than 1 kHz. For example, the max duration you can use for a
293 // frequency of 10 kHz is 6553 ms. If you use a duration longer than this,
294 // you will cause an integer overflow that produces unexpected behavior.
295 void PololuBuzzer::playNote(unsigned char note, unsigned int dur,
296  unsigned char volume)
297 {
298  // note = key + octave * 12, where 0 <= key < 12
299  // example: A4 = A + 4 * 12, where A = 9 (so A4 = 57)
300  // A note is converted to a frequency by the formula:
301  // Freq(n) = Freq(0) * a^n
302  // where
303  // Freq(0) is chosen as A4, which is 440 Hz
304  // and
305  // a = 2 ^ (1/12)
306  // n is the number of notes you are away from A4.
307  // One can see that the frequency will double every 12 notes.
308  // This function exploits this property by defining the frequencies of the
309  // 12 lowest notes allowed and then doubling the appropriate frequency
310  // the appropriate number of times to get the frequency for the specified
311  // note.
312 
313  // if note = 16, freq = 41.2 Hz (E1 - lower limit as freq must be >40 Hz)
314  // if note = 57, freq = 440 Hz (A4 - central value of ET Scale)
315  // if note = 111, freq = 9.96 kHz (D#9 - upper limit, freq must be <10 kHz)
316  // if note = 255, freq = 1 kHz and buzzer is silent (silent note)
317 
318  // The most significant bit of freq is the "divide by 10" bit. If set,
319  // the units for frequency are .1 Hz, not Hz, and freq must be divided
320  // by 10 to get the true frequency in Hz. This allows for an extra digit
321  // of resolution for low frequencies without the need for using floats.
322 
323  unsigned int freq = 0;
324  unsigned char offset_note = note - 16;
325 
326  if (note == SILENT_NOTE || volume == 0)
327  {
328  freq = 1000; // silent notes => use 1kHz freq (for cycle counter)
329  playFrequency(freq, dur, 0);
330  return;
331  }
332 
333  if (note <= 16)
334  offset_note = 0;
335  else if (offset_note > 95)
336  offset_note = 95;
337 
338  unsigned char exponent = offset_note / 12;
339 
340  // frequency table for the lowest 12 allowed notes
341  // frequencies are specified in tenths of a Hertz for added resolution
342  switch (offset_note - exponent * 12) // equivalent to (offset_note % 12)
343  {
344  case 0: // note E1 = 41.2 Hz
345  freq = 412;
346  break;
347  case 1: // note F1 = 43.7 Hz
348  freq = 437;
349  break;
350  case 2: // note F#1 = 46.3 Hz
351  freq = 463;
352  break;
353  case 3: // note G1 = 49.0 Hz
354  freq = 490;
355  break;
356  case 4: // note G#1 = 51.9 Hz
357  freq = 519;
358  break;
359  case 5: // note A1 = 55.0 Hz
360  freq = 550;
361  break;
362  case 6: // note A#1 = 58.3 Hz
363  freq = 583;
364  break;
365  case 7: // note B1 = 61.7 Hz
366  freq = 617;
367  break;
368  case 8: // note C2 = 65.4 Hz
369  freq = 654;
370  break;
371  case 9: // note C#2 = 69.3 Hz
372  freq = 693;
373  break;
374  case 10: // note D2 = 73.4 Hz
375  freq = 734;
376  break;
377  case 11: // note D#2 = 77.8 Hz
378  freq = 778;
379  break;
380  }
381 
382  if (exponent < 7)
383  {
384  freq = freq << exponent; // frequency *= 2 ^ exponent
385  if (exponent > 1) // if the frequency is greater than 160 Hz
386  freq = (freq + 5) / 10; // we don't need the extra resolution
387  else
388  freq += DIV_BY_10; // else keep the added digit of resolution
389  }
390  else
391  freq = (freq * 64 + 2) / 5; // == freq * 2^7 / 10 without int overflow
392 
393  if (volume > 15)
394  volume = 15;
395  playFrequency(freq, dur, volume); // set buzzer this freq/duration
396 }

407 // Plays the specified sequence of notes. If the play mode is
408 // PLAY_AUTOMATIC, the sequence of notes will play with no further
409 // action required by the user. If the play mode is PLAY_CHECK,
410 // the user will need to call playCheck() in the main loop to initiate
411 // the playing of each new note in the sequence. The play mode can
412 // be changed while the sequence is playing.
413 // This is modeled after the PLAY commands in GW-BASIC, with just a
414 // few differences.
415 //
416 // The notes are specified by the characters C, D, E, F, G, A, and
417 // B, and they are played by default as "quarter notes" with a
418 // length of 500 ms. This corresponds to a tempo of 120
419 // beats/min. Other durations can be specified by putting a number
420 // immediately after the note. For example, C8 specifies C played as an
421 // eighth note, with half the duration of a quarter note. The special
422 // note R plays a rest (no sound).
423 //
424 // Various control characters alter the sound:
425 // '>' plays the next note one octave higher
426 //
427 // '<' plays the next note one octave lower
428 //
429 // '+' or '#' after a note raises any note one half-step
430 //
431 // '-' after a note lowers any note one half-step
432 //
433 // '.' after a note "dots" it, increasing the length by
434 // 50%. Each additional dot adds half as much as the
435 // previous dot, so that "A.." is 1.75 times the length of
436 // "A".
437 //
438 // 'O' followed by a number sets the octave (default: O4).
439 //
440 // 'T' followed by a number sets the tempo (default: T120).
441 //
442 // 'L' followed by a number sets the default note duration to
443 // the type specified by the number: 4 for quarter notes, 8
444 // for eighth notes, 16 for sixteenth notes, etc.
445 // (default: L4)
446 //
447 // 'V' followed by a number from 0-15 sets the music volume.
448 // (default: V15)
449 //
450 // 'MS' sets all subsequent notes to play staccato - each note is played
451 // for 1/2 of its allotted time, followed by an equal period
452 // of silence.
453 //
454 // 'ML' sets all subsequent notes to play legato - each note is played
455 // for its full length. This is the default setting.
456 //
457 // '!' resets all persistent settings to their defaults.
458 //
459 // The following plays a c major scale up and back down:
460 // play("L16 V8 cdefgab>cbagfedc");
461 //
462 // Here is an example from Bach:
463 // play("T240 L8 a gafaeada c+adaeafa <aa<bac#ada c#adaeaf4");
464 void PololuBuzzer::play(const char *notes)
465 {
466  DISABLE_TIMER_INTERRUPT(); // prevent this from being interrupted
467  buzzerSequence = notes;
468  use_program_space = 0;
469  staccato_rest_duration = 0;
470  nextNote(); // this re-enables the timer interrupt
471 }

524 // Returns the numerical argument specified at buzzerSequence[0] and
525 // increments sequence to point to the character immediately after the
526 // argument.
527 static unsigned int getNumber()
528 {
529  unsigned int arg = 0;
530 
531  // read all digits, one at a time
532  char c = currentCharacter();
533  while(c >= '0' && c <= '9')
534  {
535  arg *= 10;
536  arg += c-'0';
537  buzzerSequence ++;
538  c = currentCharacter();
539  }
540 
541  return arg;
542 }

544 static void nextNote()
545 {
546  unsigned char note = 0;
547  unsigned char rest = 0;
548  unsigned char tmp_octave = octave; // the octave for this note
549  unsigned int tmp_duration; // the duration of this note
550  unsigned int dot_add;
551 
552  char c; // temporary variable
553 
554  // if we are playing staccato, after every note we play a rest
555  if(staccato && staccato_rest_duration)
556  {
557  PololuBuzzer::playNote(SILENT_NOTE, staccato_rest_duration, 0);
558  staccato_rest_duration = 0;
559  return;
560  }
561 
562  parse_character:
563 
564  // Get current character
565  c = currentCharacter();
566  buzzerSequence ++;
567 
568  // Interpret the character.
569  switch(c)
570  {
571  case '>':
572  // shift the octave temporarily up
573  tmp_octave ++;
574  goto parse_character;
575  case '<':
576  // shift the octave temporarily down
577  tmp_octave --;
578  goto parse_character;
579  case 'a':
580  note = NOTE_A(0);
581  break;
582  case 'b':
583  note = NOTE_B(0);
584  break;
585  case 'c':
586  note = NOTE_C(0);
587  break;
588  case 'd':
589  note = NOTE_D(0);
590  break;
591  case 'e':
592  note = NOTE_E(0);
593  break;
594  case 'f':
595  note = NOTE_F(0);
596  break;
597  case 'g':
598  note = NOTE_G(0);
599  break;
600  case 'l':
601  // set the default note duration
602  note_type = getNumber();
603  duration = whole_note_duration/note_type;
604  goto parse_character;
605  case 'm':
606  // set music staccato or legato
607  if(currentCharacter() == 'l')
608  staccato = false;
609  else
610  {
611  staccato = true;
612  staccato_rest_duration = 0;
613  }
614  buzzerSequence ++;
615  goto parse_character;
616  case 'o':
617  // set the octave permanently
618  octave = getNumber();
619  tmp_octave = octave;
620  goto parse_character;
621  case 'r':
622  // Rest - the note value doesn't matter.
623  rest = 1;
624  break;
625  case 't':
626  // set the tempo
627  whole_note_duration = 60*400/getNumber()*10;
628  duration = whole_note_duration/note_type;
629  goto parse_character;
630  case 'v':
631  // set the volume
632  volume = getNumber();
633  goto parse_character;
634  case '!':
635  // reset to defaults
636  octave = 4;
637  whole_note_duration = 2000;
638  note_type = 4;
639  duration = 500;
640  volume = 15;
641  staccato = 0;
642  // reset temp variables that depend on the defaults
643  tmp_octave = octave;
644  tmp_duration = duration;
645  goto parse_character;
