80  {
81  sum += analogRead(A1);

83 
84  // VBAT = 3 * millivolt reading = 3 * raw * 5000/1024
85  // = raw * 1875 / 128
86  // The correction term below makes it so that we round to the
87  // nearest whole number instead of always rounding down.
