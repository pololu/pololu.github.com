def disassemble(address)
  begin
    instr = @disasm.Disassemble(address, nil, nil)
  rescue Java::ComMicrochipMplabMdbcoreDisasm::InvalidInstructionException
    # The instruction is invalid.
    return :invalid
  end
  MplabInstruction.new instr
end
