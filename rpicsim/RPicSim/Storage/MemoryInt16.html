def value
  val = @memory.read_byte(@address) + 256 * @memory.read_byte(@address + 1)
  val -= 0x10000 if val >= 0x8000
  val
end

def value=(val)
  check_value val, -0x8000...0x8000
  @memory.write_byte(@address, val & 0xFF)
  @memory.write_byte(@address + 1, (val >> 8) & 0xFF)
end
