def value=(val)
  check_value val, 0...0x100000000
  @memory.write_byte(@address, val & 0xFF)
  @memory.write_byte(@address + 1, (val >> 8) & 0xFF)
  @memory.write_byte(@address + 2, (val >> 16) & 0xFF)
  @memory.write_byte(@address + 3, (val >> 24) & 0xFF)
end
